     // 1. 常量函数
     const fn gcd(a: u32, b: u32) -> u32 {
          match (a, b) {
               (x, 0) | (0, x) => x, //
               (x, y) if x % 2 == 0 && y % 2 == 0 => 2 * gcd(x / 2, y / 2),
               (x, y) | (y, x) if x % 2 == 0 => gcd(x / 2, y / 2),
               (x, y) if x < y => gcd((y - x) / 2, x),
               (x, y) => gcd((x - y) / 2, y),
          }
     }

     const GCD: u32 = gcd(21, 7);
