结构体内存对齐方式

   // 推断结构体占12字节
    // #[repr(C)] //使用属性不让编译器自动优化布局
    struct A {
        a: u8,  // 占1字节,按照4字节对齐，补3
        b: u32, // 占4字节，补0
        c: u16, //占2字节，补2
    }

    // 实际优化,字段重排
    struct B {
        b: u32,
        c: u16,
        d: u8,
    }

    println!("{:?}", std::mem::size_of::<A>());
    println!("{:?}", std::mem::size_of::<B>());
